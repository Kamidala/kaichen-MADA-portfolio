# Model Specification
tune_spec <- 
  decision_tree(
    cost_complexity = tune(),

  ) %>%
  set_engine("rpart") %>%
  set_mode("regression")

  cost_complexity(),
  tree_depth(),

  add_formula(BodyTemp ~.)

  tune_grid(resamples = post_cv, grid = tree_grid)
